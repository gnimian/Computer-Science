35static int
36cp1251_mbtowc (conv_t conv, ucs4_t *pwc, const unsigned char *s, int n)
37{
38 unsigned char c = *s;
39 if (c < 0x80) {
40 *pwc = (ucs4_t) c;
41 return 1;
42 }
43 else {
44 unsigned short wc = cp1251_2uni[c-0x80];
45 if (wc != 0xfffd) {
46 *pwc = (ucs4_t) wc;
47 return 1;
48 }
49 }
50 return RET_ILSEQ;
51}

91static int
92cp1251_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
93{
94 unsigned char c = 0;
95 if (wc < 0x0080) {
96 *r = wc;
97 return 1;
98 }
99 else if (wc >= 0x00a0 && wc < 0x00c0)
100 c = cp1251_page00[wc-0x00a0];
101 else if (wc >= 0x0400 && wc < 0x0498)
102 c = cp1251_page04[wc-0x0400];
103 else if (wc >= 0x2010 && wc < 0x2040)
104 c = cp1251_page20[wc-0x2010];
105 else if (wc == 0x20ac)
106 c = 0x88;
107 else if (wc == 0x2116)
108 c = 0xb9;
109 else if (wc == 0x2122)
110 c = 0x99;
111 if (c != 0) {
112 *r = c;
113 return 1;
114 }
115 return RET_ILSEQ;
116}
