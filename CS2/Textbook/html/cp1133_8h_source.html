73static int
74cp1133_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
75{
76 unsigned char c = 0;
77 if (wc < 0x00a0) {
78 *r = wc;
79 return 1;
80 }
81 else if (wc >= 0x00a0 && wc < 0x00b0)
82 c = cp1133_page00[wc-0x00a0];
83 else if (wc >= 0x0e80 && wc < 0x0ee0)
84 c = cp1133_page0e[wc-0x0e80];
85 else if (wc == 0x20ad)
86 c = 0xdf;
87 if (c != 0) {
88 *r = c;
89 return 1;
90 }
91 return RET_ILSEQ;
92}
