78static int
79iso8859_9e_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
80{
81 unsigned char c = 0;
82 if (wc < 0x00a0) {
83 *r = wc;
84 return 1;
85 }
86 else if (wc >= 0x00a0 && wc < 0x0100)
87 c = iso8859_9e_page00[wc-0x00a0];
88 else if (wc >= 0x0118 && wc < 0x01a0)
89 c = iso8859_9e_page01[wc-0x0118];
90 else if (wc >= 0x01d0 && wc < 0x01e8)
91 c = iso8859_9e_page01_d[wc-0x01d0];
92 else if (wc == 0x0259)
93 c = 0xe6;
94 else if (wc == 0x0275)
95 c = 0xf8;
96 else if (wc == 0x20ac)
97 c = 0xa4;
98 if (c != 0) {
99 *r = c;
100 return 1;
101 }
102 return RET_ILSEQ;
