30iso8859_5_mbtowc (conv_t conv, ucs4_t *pwc, const unsigned char *s, int n)
31{
32 unsigned char c = *s;

34 *pwc = (ucs4_t) c;

36 *pwc = (ucs4_t) iso8859_5_2uni[c-0xa0];

61static int
62iso8859_5_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
63{
64 unsigned char c = 0;
65 if (wc < 0x00a0) {
66 *r = wc;
67 return 1;
68 }
69 else if (wc >= 0x00a0 && wc < 0x00b0)
70 c = iso8859_5_page00[wc-0x00a0];
71 else if (wc >= 0x0400 && wc < 0x0460)
72 c = iso8859_5_page04[wc-0x0400];
73 else if (wc == 0x2116)
74 c = 0xf0;
75 if (c != 0) {
76 *r = c;
77 return 1;
78 }
79 return RET_ILSEQ;
80}
